<?php

namespace MODULE\InputFilter;

use Zend\Filter\StringTrim;
use Zend\InputFilter\InputFilter;

class FORMInputFilter extends InputFilter
{
    public function init()
    {
        $this->add([
            'name' => 'name',
            'required' => true,
            'filters' => [
                ['name' => StringTrim::class],
            ],
        ]);
    }
}
